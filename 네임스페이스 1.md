#include <iostream>

// a.cpp
namespace Ta {
	int nData;
	void PrintData()
	{
		std::cout << nData << std::endl;
	}


	void Increase(int n1)
	{
		nData += n1;
	}
}

// b.cpp
namespace Tb {
	int nData;
	void PrintData()
	{
		std::cout << "nData : " << nData
			<< std::endl;
	}
	void Increase(int n)
	{
		for (int i = 1; i <= n; i++)
			nData *= 2;
	}
}

int main()
{
	Ta::nData = 30;
	Ta::PrintData();
	Ta::Increase(10);
	Tb::nData = 20;
	Tb::PrintData();
	Tb::Increase(20);
	return 0;
}
