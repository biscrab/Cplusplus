#include <iostream>
using namespace std;
int main()
{
	{
		// Block scope 
		int a = 0;
		cout << a << endl;
		cout << &a << endl;
		cout << endl;
		{
			int a = 10; // 블록에 a가 선언되어 상단에서 선언된 a가 가렸음
			cout << a << endl;
			cout << &a << endl;
			cout << endl;
		}

		cout << a << endl;
		cout << &a << endl;
		cout << endl;
	}
	return 0;
}
