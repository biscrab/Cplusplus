#include <iostream>
using namespace std;

// 전역(개념상 무소속)
void TestFunc() {
	cout << "::TestFunc()" << endl;
}

namespace TEST
{
	// TEST namespace 소속
	void TestFunc()
	{
		cout << "TEST::TestFunc()" << endl;
	}
}
namespace MYDATA
{
	// MYDATA namespace 소속
	void TestFunc()
	{
		cout << "MYDATA::TestFunc()" << endl;
	}
}
int main()
{
	TestFunc();		// 묵시적 전역
	::TestFunc();	// 명시적 전역
	TEST::TestFunc();
	MYDATA::TestFunc();
	return 0;
}
